TRIGGER "PIPELINE"."CONTROLE_EVOLUTION_JEU"
    BEFORE INSERT
    ON jeu_match
    FOR EACH ROW

    DECLARE
        score SCORE_JEU;
        var_id_jeu NUMBER;
    BEGIN
        SELECT ID_JEU INTO var_id_jeu FROM JEU_MATCH WHERE ID_SET = :new.ID_SET AND ROWNUM = 1 ORDER BY ID_JEU DESC;
        score := RECUPERER_SCORE_JEU(var_id_jeu);
        IF (score.jeu_1 - score.jeu_2) > 1 OR (score.jeu_2 - score.jeu_1) > 1 THEN
            IF score.jeu_1 > 5 OR score.jeu_2 > 5 THEN
                INSERT_NOUVEAU_SET(var_id_jeu);
            END IF;
        END IF;
    END;
